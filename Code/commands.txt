Compile regular C file: riscv-none-elf-gcc -o output.elf -march=rv32i -mabi=ilp32 -T link.ld main.c bootloader.s -ffreestanding -nostartfiles

Compile with no STD lib: riscv-none-elf-gcc -o output.elf -march=rv32i -mabi=ilp32 -T link.ld main.c bootloader-nostd.s -ffreestanding -nostartfiles -nostdlib

riscv-none-elf-objcopy -O ihex output.elf output.hex
